   864→    function updateHourHistogram(offsetFromPT) {
   865→      const periods = [
   866→        { label: "Morning (6-12)", range: [6,7,8,9,10,11] },
   867→        { label: "Afternoon (12-18)", range: [12,13,14,15,16,17] },

   870→      ];
   871→      const adjustedCounts = {};
   872→      for (const [hour, count] of Object.entries(rawHourCounts)) {
   873→        const newHour = (parseInt(hour) + offsetFromPT + 24) % 24;
   874→        adjustedCounts[newHour] = (adjustedCounts[newHour] || 0) + count;
   875→      }
   876→      const periodCounts = periods.map(p => ({
   877→        label: p.label,
   878→        count: p.range.reduce((sum, h) => sum + (adjustedCounts[h] || 0), 0)
   879→      }));
   880→      const maxCount = Math.max(...periodCounts.map(p => p.count)) || 1;
   881→      const container = document.getElementById('hour-histogram');
   882→      container.textContent = '';
   883→      periodCounts.forEach(p => {
   884→        const row = document.createElement('div');
   885→        row.className = 'bar-row';
   886→        const label = document.createElement('div');
   887→        label.className = 'bar-label';
   888→        label.textContent = p.label;
   889→        const track = document.createElement('div');
   890→        track.className = 'bar-track';
   891→        const fill = document.createElement('div');
   892→        fill.className = 'bar-fill';
   893→        fill.style.width = (p.count / maxCount) * 100 + '%';

   899→        row.appendChild(label);

   902→        container.appendChild(row);
   903→      });
   904→    }
